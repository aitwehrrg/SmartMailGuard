{ (x<sup>1</sup>, y<sup>1</sup>), (x<sup>2</sup>, y<sup>2</sup>), ... , }

train set = m<sub>train</sub>

test set = m<sub>test</sub>

X = [ x<sup>1</sup>, x<sup>2</sup>, ... , x<sup>m</sup>]

Y = [ y<sup>1</sup>, y<sup>2</sup>, ... , y<sup>m</sup>]
